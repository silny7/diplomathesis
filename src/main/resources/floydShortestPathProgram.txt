Program shortestpath
declare
    n,k: integer;
    D: array [0..9, 0..9] of integer
initially
    n := 10[]
    k := 0 []
    <<|| i,j : (0<=i<n) and (0<=j<n) :: D[i,j] := Random(0,20) >>
assign
   <<|| i,j : (0<=i<n) and (0<=j<n) ::
        D[i,j] := Min(D[i,j], D[i,k] + D[k,j]) 
    >> 
   ||  k := k + 1 if k < n - 1
end
